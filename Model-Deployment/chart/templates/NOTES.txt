1. Confirm the model deployment is running:
  kubectl get pods --namespace {{ .Release.Namespace }} -l "app.kubernetes.io/instance={{ .Release.Name }}"

2. Get the application URL by running these commands:
{{- if .Values.ingress.enabled }}
{{- range $host := .Values.ingress.hosts }}
  {{- range .paths }}
  http{{ if $.Values.ingress.tls }}s{{ end }}://{{ $host.host }}{{ .path }}
  {{- end }}
{{- end }}
{{- else if contains "NodePort" .Values.service.type }}
  export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "model-deployment.fullname" . }})
  export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
  echo http://$NODE_IP:$NODE_PORT
{{- else if contains "LoadBalancer" .Values.service.type }}
     NOTE: It may take a few minutes for the LoadBalancer IP to be available.
           You can watch its status by running 'kubectl get --namespace {{ .Release.Namespace }} svc -w {{ include "model-deployment.fullname" . }}'
  export SERVICE_IP=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ include "model-deployment.fullname" . }} --template "{{"{{ range (index .status.loadBalancer.ingress 0) }}{{.}}{{ end }}"}}")
  echo http://$SERVICE_IP:{{ .Values.service.port }}
{{- else if contains "ClusterIP" .Values.service.type }}
  echo "Visit http://127.0.0.1:{{ .Values.service.port }} to use your application"
  kubectl --namespace {{ .Release.Namespace }} port-forward svc/{{ include "model-deployment.fullname" . }} {{ .Values.service.port }}:{{ .Values.service.port }}
{{- end }}

3. Inspect the model-specific environment in the pod:
  kubectl exec --namespace {{ .Release.Namespace }} deploy/{{ include "model-deployment.fullname" . }} -- printenv MODEL_NAME MODEL_PATH

4. Security baseline (defaults):
  - runAsNonRoot: {{ .Values.podSecurityContext.runAsNonRoot }}
  - readOnlyRootFilesystem: {{ .Values.securityContext.readOnlyRootFilesystem }}
{{- if .Values.securityContext.readOnlyRootFilesystem }}

  IMPORTANT: the root filesystem is READ-ONLY. A writable emptyDir is mounted
  at /tmp (writableTmp.enabled={{ .Values.writableTmp.enabled }}). If your container writes to
  any other path, the process will fail. Add those paths via volumes/volumeMounts,
  or set securityContext.readOnlyRootFilesystem=false.
{{- end }}
